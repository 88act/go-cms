|   |   └──  .js --  前端控制库

|   |   ├── getters.js -- getters

|   ├── styles -- css
|   ├── utils -- utils 组件

|   |   ├── dashboard -- 面板
|   |   ├── error -- 错误

|   |   ├── init -- 初始化数据  
|   |   |   ├── index -- 新版本
|   |   |   ├── init -- 旧版本
|   |   ├── layout  --  layout约束页面 
|   |   |   ├── aside -- 
|   |   |   ├── bottomInfo -- bottomInfo
